class Appointment(models.Model):
    person = models.ForeignKey(Person, on_delete=models.CASCADE)
    title = models.CharField(max_length=100)
    details = models.CharField(blank=True, max_length=1000)
    appointment = models.ForeignKey(Person, blank=True, null=True, related_name='schedule', on_delete=models.SET_NULL)
    time = models.DateTimeField(null=True, blank=True)
    STATUS_CHOICES = (
        ('Pending', 'Pending'),
        ('Done', 'Done'),
        ('Canceled', 'Canceled'),
    )
    status = models.CharField(max_length=20, choices=STATUS_CHOICES)


class Schedule(models.Model):
    person = models.ForeignKey(Person, on_delete=models.CASCADE)
    title = models.CharField(max_length=100)
    details = models.CharField(blank=True, max_length=1000)
    appointment = models.ForeignKey(Appointment, blank=True, null=True, related_name='schedule', on_delete=models.SET_NULL)
    time = models.DateTimeField(null=True, blank=True)
    STATUS_CHOICES = (
        ('Pending', 'Pending'),
        ('Done', 'Done'),
        ('Canceled', 'Canceled'),
    )
    status = models.CharField(max_length=20, choices=STATUS_CHOICES)
